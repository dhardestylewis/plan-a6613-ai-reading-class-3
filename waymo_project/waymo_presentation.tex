\section{Part 3: Policy Recommendations}

\begin{frame}{The Core Challenge}
\begin{center}
\Huge{\textbf{Integration without Disruption}}
\vspace{1.5cm}

\Large{How do we maximize safety without exacerbating spatial inequities or cannibalizing public transit?}
\end{center}
\end{frame}

\begin{frame}{Policy Interventions}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\Large{\textbf{1. Mobility Equity Framework}}
\vspace{0.3cm}
\normalsize
\begin{itemize}
    \item Condition operational permits on serving transit-desert neighborhoods.
    \item Proportional service expansion.
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\Large{\textbf{2. Grid Integration \& V2I}}
\vspace{0.3cm}
\normalsize
\begin{itemize}
    \item Mandate off-peak charging depots.
    \item Upgrade traffic signals for Vehicle-to-Infrastructure (V2I) communication.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{3. AV-Transit Hubs}
\begin{columns}[c]
\begin{column}{0.4\textwidth}
\Large{\textbf{First/Last Mile Connectivity}}
\vspace{0.7cm}

\normalsize
\begin{itemize}
    \item Dedicated, geofenced pick-up zones at stations.
    \item Favorable pricing for rides connecting directly to transit.
\end{itemize}
\end{column}
